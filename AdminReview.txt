<script>

function approveTask(){

  const btn =
    document.getElementById(
      'approveBtn'
    );

  btn.disabled = true;

  btn.innerHTML =
    '⏳ Approving...';

  google.script.run

    .withSuccessHandler(
      function(){

        btn.innerHTML =
          '✅ Approved';

        alert(
          'Task Approved Successfully'
        );

      }
    )

    .withFailureHandler(
      function(error){

        btn.disabled = false;

        btn.innerHTML =
          'APPROVE TASK';

        alert(
          error.message
        );

      }
    )

    .approveTask(
      '<?= taskId ?>'
    );

}

function rejectTask(){

  const comment =
    prompt(
      'Enter rejection comment'
    );

  if(

    !comment ||

    comment.length < 10

  ){

    alert(
      'Minimum 10 characters required'
    );

    return;

  }

  const btn =
    document.getElementById(
      'rejectBtn'
    );

  btn.disabled = true;

  btn.innerHTML =
    '⏳ Rejecting...';

  google.script.run

    .withSuccessHandler(
      function(){

        btn.innerHTML =
          '❌ Rejected';

        alert(
          'Task Rejected Successfully'
        );

      }
    )

    .withFailureHandler(
      function(error){

        btn.disabled = false;

        btn.innerHTML =
          'REJECT TASK';

        alert(
          error.message
        );

      }
    )

    .rejectTask(

      '<?= taskId ?>',

      comment

    );

}

</script>
