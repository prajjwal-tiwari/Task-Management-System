<script>

function createTask(){

  const btn =
    document.getElementById(
      'createBtn'
    );

  const obj = {

    project:
      document.getElementById('project').value.trim(),

    model:
      document.getElementById('model').value.trim(),

    category:
      document.getElementById('category').value.trim(),

    priority:
      document.getElementById('priority').value,

    taskTitle:
      document.getElementById('taskTitle').value.trim(),

    description:
      document.getElementById('description').value.trim(),

    employeeName:
      document.getElementById('employeeName').value.trim(),

    employeeEmail:
      document.getElementById('employeeEmail').value.trim(),

    ccEmail:
      document.getElementById('ccEmail').value.trim(),

    targetDate:
      document.getElementById('targetDate').value,

    targetTime:
      document.getElementById('targetTime').value,

    reminderType:
      document.getElementById('reminderType').value

  };

  if(

    !obj.project ||
    !obj.taskTitle ||
    !obj.employeeName ||
    !obj.employeeEmail ||
    !obj.targetDate

  ){

    alert(
      'Please fill all mandatory fields'
    );

    return;

  }

  btn.disabled = true;

  btn.innerHTML =
    '⏳ Creating Task...';

  google.script.run

    .withSuccessHandler(
      function(res){

        alert(
         'Task Created : ' +
         res.taskId

        );

        btn.innerHTML =
          '✅ Task Created';

      }
    )

    .withFailureHandler(
      function(error){

        btn.disabled = false;

        btn.innerHTML =
          'CREATE TASK';

        alert(
          error.message
        );

      }
    )

    .createTask(obj);

}

</script>
