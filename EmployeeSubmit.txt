<script>

function submitTask(){

  const remark =

    document
      .getElementById(
        'remark'
      )
      .value
      .trim();

  if(
    remark.length < 10
  ){

    alert(
      'Please enter proper remark'
    );

    return;
  }

  const btn =
    document.getElementById(
      'submitBtn'
    );

  btn.disabled = true;

  btn.innerHTML =
    '⏳ Submitting...';

  google.script.run

    .withSuccessHandler(
      function(){

        btn.innerHTML =
          '✓ Submitted';

        alert(
          'Task Submitted Successfully'
        );

      }
    )

    .withFailureHandler(
      function(error){

        btn.disabled = false;

        btn.innerHTML =
          'SUBMIT TASK';

        alert(
          error.message
        );

      }
    )

    .submitTask(

      '<?= taskId ?>',

      remark

    );

}


</script>
